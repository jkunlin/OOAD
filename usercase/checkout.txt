checkout命令：切换分支（或版本）并设置当前工作目录内容，或仅更改当前工作目录内容。

用户输入checkout <filepath|commit_id|tag|branch>
		if（输入的是commit_id）{
			if (当前暂存区是干净的){
				跳转到commit_id所对应的版本
				将当前工作目录设置成该版本对应的内容
			}
			else {
				提示用户先处理暂存区内容，或输入强制checkout命令，抛弃暂存区内容
				用况结束
			}
		}
		if（输入的是tag）{
			if (当前暂存区是干净的){
				跳转到tag所对应的版本
				将当前工作目录设置成该版本对应的内容
			}
			else {
				提示用户先处理暂存区内容，或输入强制checkout命令，抛弃暂存区内容
				用况结束
			}
		}
		if（输入的是filepath）{
			if(<filepath>是普通文件){
				<filepath>恢复到当前版本的原始状态
			}
			if（<filepath>是目录文件）{
				递归地将<filepath>下的所有文件恢复到当前版本的状态
				
			}
		}
		if（输入的是branch）{
			if (当前暂存区是干净的) {
				跳转到branch分支
				将当前工作目录设置成该branch最新版本对应的内容
			}
			else {
				提示用户先处理暂存区内容，或输入强制checkout命令，抛弃暂存区内容
				用况结束
			}
		}
